        function filterCategory(category, element) {
            currentCategory = category;
            document.querySelectorAll('.category-tab').forEach(tab => tab.classList.remove('active'));
            element.classList.add('active');
            renderProducts();
        }

        // --- LÓGICA DA REALIDADE AUMENTADA ---

        function showError(title, message) {
            arVideo.style.display = 'none';
            arError.style.display = 'flex';
            arErrorTitle.textContent = title;
            arErrorMessage.textContent = message;
        }

        async function openArView(productId) {
            const product = PRODUCTS.find(p => p.id === productId);
            if (!product) return;

            // Prepara a interface de AR
            document.body.classList.add('ar-active');
            arProductName.textContent = product.name;
            arOverlayImage.src = product.pngImage;
            arView.classList.add('visible');
            arVideo.style.display = 'block';
            arError.style.display = 'none';

            // Verifica se o navegador suporta a API
            if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
                showError('Função não suportada', 'Seu navegador não suporta o acesso à câmera. Tente usar um navegador moderno como Chrome ou Safari.');
                return;
            }

            try {
                // Pede permissão e inicia a câmera traseira
                const stream = await navigator.mediaDevices.getUserMedia({
                    video: {
                        facingMode: 'environment' // 'environment' para traseira, 'user' para frontal
                    }
                });
                
                currentStream = stream;
                arVideo.srcObject = stream;

            } catch (err) {
                // Trata erros comuns de permissão
                console.error("Erro ao acessar a câmera: ", err);
                if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
                    showError('Acesso negado', 'Você precisa permitir o acesso à câmera para usar esta função. Verifique as permissões do site nas configurações do seu navegador.');
                } else if (err.name === 'NotFoundError' || err.name === 'DevicesNotFoundError') {
                    showError('Câmera não encontrada', 'Não foi possível encontrar uma câmera traseira no seu dispositivo.');
                } else {
                    showError('Ocorreu um erro', 'Não foi possível iniciar a câmera. Por favor, tente novamente.');
                }
            }
        }

        function closeArView() {
            // Para o stream da câmera para economizar bateria e preservar a privacidade
            if (currentStream) {
                currentStream.getTracks().forEach(track => {
                    track.stop();
                });
            }
            arVideo.srcObject = null;
            currentStream = null;

            // Esconde a interface de AR
            arView.classList.remove('visible');
            document.body.classList.remove('ar-active');
        }
